#include <iostream>

using namespace std;

int main()
{
  const double a = 4.1;
  const double b = 2.7;
  for (double x = 1.5; x <= 3.5; x += 0.4)
  {
      double numenator = a * pow(x, 1 / 3.0) - b * log(x) / log(5);
      double denominator = pow(log(x -  1)) / log(10), 3);
      double y = numenator / denominator;
      
      std::cout << "x = " << x << std::endl;
      std::cout << "y = " << y << std::endl;
  }

    return 0;
}
